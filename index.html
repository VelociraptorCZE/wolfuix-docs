Introduction

Wolfuix is a lightweight JS framework. It includes some polyfills, components and tools. Down below you can find some examples.

If you use Wolfuix alongside with the webpack, then you will ensure compatibility with IE10+.

import Wolfuix from "wolfuix/Wolfuix";

const wolfuix = new Wolfuix();

const button = new wolfuix.component.button("button-id");

button.props = {
    value: 0,
    target: button.el.getElem("elem-to-print"),
    clickHandler: () => {
        button.props.value++;
        button.props.target.innerText = button.props.value;
    }
};
// Lower on this page this example is available
            
const array = [
    ["prop", 664],
    ["anotherProp", "anotherValue"]
];

const obj = array.toObject(); // returns {prop: 664, anotherProp: "anotherValue"}
            
import WolfuixFormData from "wolfuix/js/lib/WolfuixFormData";

const formData = new WolfuixFormData(); // init the instance without any data
formData.append("name", "value");
formData.append("hours", 16);

let serialized = formData.serialize(); // name=value&hours=16
let data = formData.valuesAsFormData(); // returns the instance of FormData
let json = formData.toJSON(); // {"name":"value","hours":16}
let deserialized = WolfuixFormData.deserialize("name=%24value&surname=!%26%24content"); // [["name", "$value"], ["surname", "!&$content"]]

console.log(serialized, data, json, deserialized)
